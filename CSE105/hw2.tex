\textbf{b.} The following function returns the set of all reachable states, given a DFA M. In every iteration of the outer loop, the set of reachable states is updated. Furthermore, once a state is denoted as reacheble, it will never be visited again, since it will be subtracted from the $new$ set. Consequently, the algorithm is guaranteed to terminate.

\begin{algorithmic}
\Function{findReachable}{DFA M}
    \State $reachable \gets \{q_0\}$
    \State $new \gets \{q_0\}$

	\Repeat 
	    \State $temp \gets \{\}$
	    \For {each $q \in new$}
	    	\For {all $c \in \Sigma$}
	    		\State $temp \gets temp \cup [p \mid p = \delta(q,c)]$
	    	\EndFor
	    \EndFor
	    \State $new \gets temp \setminus reachable$
	    \State $reachable \gets reachable \cup new$
	\Until {$new$ $!=$ $\{\}$}    
    
    \State \Return $reachable$
\EndFunction
\end{algorithmic}
